struct BiTree{
     char data;
     struct BiTree *lchild;
     struct BiTree *rchild;
     BiTree(char c):data(c),lchild(NULL),rchild(NULL){}
     BiTree(char c,BiTree *L,BiTree *R):data(c),lchild(L),rchild(R){}
};

//DFS求二叉树高度
int maxdeep(BiTree *root)
{
     if(root == NULL)return 0;
     return max(maxdeep(root->lchild),maxdeep(root->rchild))+1;
}
//DFS判断平衡二叉树高度是否平衡
bool isbalance(BiTree *root)
{
     if(root == NULL)return true;
     return abs(maxdeep(root->lchild)-maxdeep(root->rchild))<=1 && isbalance(root->lchild) && isbalance(root->rchild);
}

BFS

void BFS(BiTree *root)
{
     queue<BiTree*>Q;
     Q.push(root);
     int level=0;
     while(!Q.empty())
     {
       int size = Q.size();
       for(int i=0;i<size;i++)
       {
       	BiTree *temp = Q.front();
       	Q.pop();
       	if(temp->lchild != NULL)Q.push(temp->lchild);
       	if(temp->rchild != NULL)Q.push(temp->rchild);
       }
       level++;
     }
}

· sort

快排

int Partition(int A[],int low,int high)
{
     int pivot = A[low];
     while(low<high)
     {
       while(low<high && A[high]>pivot)high--;
       A[low] = A[high];
       while(low<high && A[low]<pivot)low++;
       A[high] = A[low];
     }
     A[low] = pivot;
     return low;
}
void FastSort(int A[],int low,int high)
{
     if(low<high)
     {
       int pivot = Partition(A,low,high);
       FastSort(A,low,pivot-1);
       FastSort(A,pivot+1,high);
     }
}

归并

//A[]好像是伪代码
int *B = new int(high+1);
void Merge(int A[],int low,int mid,int high)
{
     for(int i=low;i<=high;i++)
     	B[i] = A[i];
     int i=low,j=mid+1,k=low;
     while(i<=mid && j<=high)
     {
     	if(A[i]<A[j])
     	{
         B[k++] = A[i++];
     	}else
     	{
         B[k++] = A[j++];
     	}
     }
     while(i<=mid)B[k++] = A[i++];
     while(j<=high)B[k++] = A[j++];
}
void MergeSort(int A[],int low,int high)
{
     if(low<high)
     {
       mid = (low+high)/2;
       MergeSort(A,low,mid);
       MergeSort(A,mid+1,high);
       Merge(A,low,mid,high);
     }
}

Bubble

void Bubble(vector<int>&v,int n)
{
     for(int i=0;i<n;i++)
     {
     	int flag = -1;
       for(int j=n-1;j>i;j--)
       {
         if(v[j]<v[j-1])
         {
           swap(v[j],v[j-1]);
           flag = 1;
         }
       }
       if(flag==-1)break;
     }
}

int a,b;
cin>>a>>b;//保证a、b< 65536，那么有效数字位数为低16位
int flag=false;
for(int i=0;i<16;i++)
{
     if(a==b)
     {
       flag=true;//a和b在旋转左移任意位后是相等的

     }
     a = (((a>>15)&1)|(a<<1))&0xffff;//a的二进制旋转左移，无须将十进制转换为二进制就可以实现
