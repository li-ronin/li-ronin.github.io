单源最短路径

/*
·给定邻接矩阵G[][]
·定义：1、int数组dis[]记录源点到各个节点最短距离。
      2、bool数组fin[]记录各节点是否已经找到最小距离
·初始化：dis[源点]=0、dis[其余]=Inf ;fin[源点]=true 、fin[其余]=false
*/
const int Inf = (1<<31)-1;
void Dijkstra(int G[][],int n)
{
     int *dis = new int(n);
     bool *fin = new bool(n);
     memset(dis,Inf,sizeof(dis));
     memset(fin,false,sizeof(fin));
  dis[0] = 0;
     //*********KeyCode***********
     for(int i=0;i<n;i++)
     {
       int u=-1,min = Inf;
       for(int j=0;j<n;j++)//找出本轮最小距离 用u暂存；
       {
         if(fin[j]==false && dis[j]<min)
         {
           u = j;
           min = dis[j];
         }
       }
       if(u=-1)break;
       fin[u] = true;
       for(int v=0;v<n;v++)//更新u邻居结点的距离
       {
         if(fin[v]==false && G[u][v]+dis[u] < dis[v])
         {
           dis[v] = G[u][v] + dis[u];
         }
       }
     }
      //*********KeyCode***********
}

Prim 最小生成树MST（min-spanning-tree）

const int inf = 1<<30;
vector<vector<int>>G(n,vector<int>(n));
   vector<int>lowcost(n);
   int sum=0;
   lowcost[0]=0;
   for(int i=1;i<n;i++)lowcost[i] = G[0][i];
   for(int i=1;i<n;i++)
   {
       int min=inf,k=-1;
       for(int j=1;j<n;j++) //找出本轮lowcost中非零的最小权值 记录在min和K中
       {
           if(lowcost[j]!=0 && lowcost[j]<min)
           {
               min = lowcost[j];
               k = j;
           }
       }
       //******************
       if(k==-1)break;
       lowcost[k] = 0;		//选中当前最小结点 K， 令K的lowcost=0
       sum+=min;
       //*******************
       for(int j=1;j<n;j++)//以K为主角去更新它的邻居节点的权值，
       {
           if(lowcost[j]!=0 && G[k][j]<lowcost[j])//Dijkstra是dis[k]+G[k][j] <dis[j]
           {
               lowcost[j]=G[k][j];
           }
       }
   }

Dijkstra与Prim的区别

·Dijkstra选择的是从源点处到当前结点的最短距离；
·Prim选择的是当前结点的到已经完结结点visit[i] = true的最短距离；

#include<iostream>
 #include<vector>
#include<math.h>
using namespace std;
const int inf = 1<<30;
int main()
{
    int n;
    cin>>n;
    vector<vector<double>>G(n,vector<double>(n));
    vector<vector<double>>coordinate(n,vector<double>(2));
    
    for(int i=0;i<n;i++)
    {
        cin>>coordinate[i][0]>>coordinate[i][1];
        for(int j=0;j<n;j++)
            G[i][j]=inf;
    }
    for(int i=0;i<n-1;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            double a=coordinate[i][0]-coordinate[j][0];
            double b=coordinate[i][1]-coordinate[j][1];
            a = pow(a,2);
            b = pow(b,2);
            G[i][j]=G[j][i]=sqrt(a+b);
        }
    }
  
    vector<double>lowcost(n);
    double sum=0;
    lowcost[0]=0;
    for(int i=1;i<n;i++)lowcost[i] = G[0][i];
    for(int i=1;i<n;i++)
    {
        double min=inf,k=-1;
        for(int j=1;j<n;j++)
        {
            if(lowcost[j]!=0 && lowcost[j]<min)
            {
                min = lowcost[j];
                k = j;
            }
        }
        if(k==-1)break;
        lowcost[k] = 0;
        sum+=min;
        for(int j=1;j<n;j++)
        {
            if(lowcost[j]!=0 && G[k][j]<lowcost[j])
            {
                lowcost[j]=G[k][j];
            }
        }
    }
    printf("%.2lf",sum);
    return 0;
}
